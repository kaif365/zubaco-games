API Reference

          


            

              
                POST
                /game/session/start
              
              

                
Request Body

                
{
  "stageId": "uuid"
}

                
Response · data

                
{
  "sessionId": "uuid",
  "stageId": "uuid",
  "status": "ACTIVE",
  "completedRounds": 0,
  "currentLevelId": "uuid",
  "totalDemoRounds": 0,
  "totalActualRounds": 3,
  "currentDemoRound": 0,
  "currentActualRound": 1,
  "requestedDemoRound": 0,
  "requestedActualRound": 1,
  "startTime": "ISO string (actual rounds only)",
  "endTime": "ISO string (actual rounds only)",
  "board": { ...FormattedBoard }
}

                POST
                /game/session/save-progress
              
              

                
Request Body

                
{
  "sessionId": "uuid",
  "board": {
    "sessionBoardId": "uuid",
    "version": 0,
    "paths": [
      {
        "color": "red",
        "moveId": "uuid",
        "timeStamp": "2024-01-01T00:00:00.000Z",
        "path": [{"row":0,"col":0},{"row":1,"col":0}]
      }
    ]
  }
}

                
Response · data

                
{ "saved": true, "ignored": false }
// or when nothing changed:
{ "saved": false, "ignored": true }

              

            


            

              
                POST
                /game/session/complete-board
              
              

                
Request Body

                
same shape as save-progress

                
Response · data (ACTIVE)

                
{
  "sessionId": "uuid",
  "stageId": "uuid",
  "status": "ACTIVE",
  "completedRounds": 2,
  "currentLevelId": "uuid",
  "currentActualRound": 3,
  "board": { ...FormattedBoard }
}

                
Response · data (COMPLETED)

                
{
  "sessionId": "uuid",
  "stageId": "uuid",
  "status": "COMPLETED",
  "finalScore": 960
}
